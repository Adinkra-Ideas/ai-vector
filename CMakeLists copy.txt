cmake_minimum_required(VERSION 3.5)

project(aivector LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(Main main.cpp)
set(AllFiles
    vector.hpp)
set(Testers
    vector_unittest.hpp)

add_executable(aivector ${Main} ${AllFiles})
if (MSVC)
  # warning level 4
  target_compile_options(aivector
        /W4)
else()
  # additional warnings
  target_compile_options(aivector PUBLIC
        -Wextra
        -Werror
        -Wall
        -Wpedantic)
endif()

add_executable(aivectorTDD ${AllFiles} ${Testers})...

include(GNUInstallDirs)
install(TARGETS aivector
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)
